from pyspark.sql import functions as F from pyspark.sql.window import Window # DataFrame transformations — ən çox istifadə olunan pattern df = (spark.read .format("delta") .load("s3://datalake/sales")) # Window function — hər kateqoriyada sıralama window_spec = Window.partitionBy("category").orderBy(F.desc("revenue")) result = (df .filter(F.col("date") >= "2024-01-01") .withColumn("rank", F.rank().over(window_spec)) .withColumn("pct_total", F.col("revenue") / F.sum("revenue").over(Window.partitionBy("category"))) .filter(F.col("rank") <= 10)) result.write.format("delta").mode("overwrite").save("s3://output/top10")

  • Column pruning

💾

Caching & Persistence

Düzgün cache strategiyası böyük fərq yaradır. Səhv cache isə memory doldurar.

  • unpersist() — məcburidir!
  • Checkpoint — lineage kəsmək
  • Nə vaxt cache etməli?

Broadcast Join

Kiçik table (<10MB) olduqda broadcast join shuffle-ı tamamilə aradan qaldırır.

# Spark UI-da plan oxumaq — kritik skill df.explain(mode="formatted") # Parsed, Analyzed, Optimized, Physical # AQE (Adaptive Query Execution) — Spark 3.0+ spark.conf.set("spark.sql.adaptive.enabled", "true") spark.conf.set("spark.sql.adaptive.coalescePartitions.enabled", "true") spark.conf.set("spark.sql.adaptive.skewJoin.enabled", "true") # Data Skew problemi — salt+repartition trick skewed_df = df.withColumn("salt", (F.rand() * 10).cast("int")) other_df_expanded = other_df.withColumn("salt", F.explode(F.array(*[F.lit(i) for i in range(10)]))) result = skewed_df.join(other_df_expanded, ["key", "salt"]) # Optimal partition sayı target_partitions = max(200, df.rdd.getNumPartitions() * 2) df.repartition(target_partitions, "date", "region")

# Kafka → Delta Lake Streaming Pipeline from pyspark.sql.functions import from_json, col from pyspark.sql.types import StructType, StringType, TimestampType schema = (StructType() .add("user_id", StringType()) .add("event", StringType()) .add("timestamp", TimestampType())) # Kafka source stream_df = (spark .readStream .format("kafka") .option("kafka.bootstrap.servers", "broker:9092") .option("subscribe", "events") .option("startingOffsets", "latest") .load()) # Parse + Watermark + Window Aggregation parsed = (stream_df .select(from_json(col("value").cast("string"), schema).alias("data")) .select("data.*") .withWatermark("timestamp", "10 minutes")) # Delta Lake sink — exactly-once semantics query = (parsed .writeStream .format("delta") .outputMode("append") .trigger(processingTime="30 seconds") .option("checkpointLocation", "s3://checkpoints/events") .start("s3://datalake/events")) query.awaitTermination()

🏅

Medallion Architecture

Lakehouse-da standart data katmanlaşdırma arxitekturası. Sənin 7 aylıq təcrübəndə bu olub!

🥉 Bronze Layer

Raw data — heç bir transformasiya, append-only, tam tarix saxlanılır. Schema-on-read.

🥈 Silver Layer

Cleaned, deduplicated, validated data. Schema enforced. Join-lər burada.

🥇 Gold Layer

Business-ready aggregated data. BI toollar buradan baxır. Yüksək query performance.

🔺

Delta Lake Deep Dive

Lakehouse-un əsası. Transaction log, ACID, time travel, schema evolution.

  • ACID transactions (concurrent writes)
  • Time travel — VERSION AS OF, TIMESTAMP AS OF
  • Schema evolution — mergeSchema
  • OPTIMIZE + Z-ORDER clustering
  • VACUUM — köhnə faylları silmək
  • Change Data Feed (CDF)
  • MERGE (upsert) — SCD Type 2
  • Deletion Vectors (Spark 3.4+)
# Delta Lake MERGE — SCD Type 2 pattern from delta.tables import DeltaTable delta_table = DeltaTable.forPath(spark, "s3://datalake/customers") (delta_table.alias("target") .merge( source=new_data.alias("source"), condition="target.customer_id = source.customer_id AND target.is_current = true" ) .whenMatchedUpdate( condition="source.updated_at > target.updated_at", set={"is_current": "false", "end_date": "source.updated_at"} ) .whenNotMatchedInsert(values={ "customer_id": "source.customer_id", "name": "source.name", "is_current": "true", "start_date": "source.updated_at" }) .execute()) # Z-Order clustering — read performance üçün spark.sql(""" OPTIMIZE delta.`s3://datalake/events` ZORDER BY (date, user_id) """)
